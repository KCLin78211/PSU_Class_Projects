% Distributions
\newcommand{\normsub}{&a \overline{Y}_{A\cdot} - b \overline{Y}_{B\cdot} \sim N(a(\mu + \tau_A), a^2 \sigma^2) + N(-b(\mu + \tau_A), (-b)^2 \sigma^2)}

\newcommand{\chisq}{\sum_{i=1}^{n} Y_i^2 \sim \chi^2_n}

\newcommand{\fdist}{&W_1 \sim \chi^2_(d_1); \quad W_2 \sim \chi^2_(d_2); \quad \frac{W_1/d_1}{W_2/d_2} = Q \sim F_{d_1,d_2}}
\newcommand{\tdist}{\iff d_1=1, \sqrt{Q} \sim T_{d_2}}


% Estimability
\newcommand{\estsum}{\sum_{i=1}^v b_i\left(\hat{\mu} + \hat{\tau}_i\right) & = \sum_{i=1}^v b_i\overline{Y}_i}

\newcommand{\ybar}{\overline{Y}_{i\cdot} = \hat{\mu}+\hat{\tau}_i = \frac{1}{r_i}\sum_{t=1}^nY_{it} \sim N\left(\mu+\tau_i,\frac{1}{r_i}\sigma^2\right)}

% Sums of Squares
\newcommand{\SSE}{\text{SSE} &= \sum_{i=1}^{v}\sum_{t=1}^{r_i} \left(\overline{Y}_{it} - \overline{Y}_{i\cdot}\right)^2}

\newcommand{\SST}{\text{SST} = \sum_{i=1}^{v} r_i \left(\overline{Y}_{i\cdot} - \overline{Y}_{\cdot\cdot}\right)^2}

\newcommand{\SSTOT}{\text{SSTOT} = \sum_{i=1}^{v}\sum_{t=1}^{r_i} \left(\overline{Y}_{it} - \overline{Y}_{i\cdot}\right)^2}

% Sigma Stuff
\newcommand{\sigmaest}{\widehat{\sigma^2} & \approx \frac{\text{SSE}}{n-v}}

\newcommand{\ssedist}{\frac{\text{SSE}}{\sigma^2} \sim \chi^2_{n-v}}

% Test Statistics
\newcommand{\fdiststat}{T^* &= \frac{\text{SST}/(v-1)}{\text{SSE}/(n-v)} \sim F_{(v-1),(n-v)}}
\newcommand{\tdiststat}{T^* &= \frac{\left(\overline{Y}_{A\cdot}-\overline{Y}_{C\cdot}\right)^2/\left(K(\sigma)\right)^2}{\text{SSE}/\left[\left(n-v\right)\sigma^2\right]} \sim F_{1,(n-v)}}
\newcommand{\tdistaltstat}{\frac{\sqrt{\frac{r_i+r_j}{r_i r_j}}\left(\overline{Y}_i - \overline{Y}_j\right)}{\sqrt{SSE/\left(n-v\right)}}\sim F_{1,(n-v)}}
